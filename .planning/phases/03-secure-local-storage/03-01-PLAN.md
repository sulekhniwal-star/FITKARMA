    <automated>mkdir -p lib/core/security && echo "import 'dart:convert';\nimport 'package:flutter_secure_storage/flutter_secure_storage.dart';\nimport 'package:hive_ce/hive.dart';\n\nclass EncryptionService {\n  static const _keyName = 'fitkarma_encryption_key';\n  final _storage = const FlutterSecureStorage();\n\n  Future&lt;List&lt;int&gt;&gt; getOrCreateKey() async {\n    final storedKey = await _storage.read(key: _keyName);\n    if (storedKey != null) return base64Decode(storedKey);\n\n    final newKey = Hive.generateSecureKey();\n    await _storage.write(key: _keyName, value: base64Encode(newKey));\n    return newKey;\n  }\n}" > lib/core/security/encryption_service.dart</automated>
